Certainly! Here's the C program using `scanf` to get input for your personal information:

```c
#include <stdio.h>

int main() {
    // Personal Information
    char name[50];
    char dob[20];
    char mobile[20];
    char email[50];
    char address[100];

    // Getting Input for Personal Information
    printf("Enter Your Name: ");
    scanf("%49s", name); // Use %49s to limit the input to 49 characters to prevent buffer overflow
    
    printf("Enter Date of Birth: ");
    scanf("%19s", dob);
    
    printf("Enter Mobile Number: ");
    scanf("%19s", mobile);
    
    printf("Enter Email ID: ");
    scanf("%49s", email);
    
    // Consume the newline character left in the input buffer
    getchar();
    
    printf("Enter Address: ");
    fgets(address, sizeof(address), stdin);

    // Displaying Personal Information
    printf("\nPersonal Information:\n");
    printf("Name: %s\n", name);
    printf("Date of Birth: %s\n", dob);
    printf("Mobile Number: %s\n", mobile);
    printf("Email ID: %s\n", email);
    printf("Address: %s", address);

    return 0;
}
```

This program uses `scanf` to get input for the name, date of birth, mobile number, and email ID while using `fgets` for the address. It limits the input size to avoid buffer overflow issues.
