#include <stdio.h>
#include <math.h>
int main() {
    int num, originalNum, remainder, result = 0;
    printf("Enter a number: ");
    scanf("%d", &num);
    originalNum = num;
    while (originalNum != 0) {
        remainder = originalNum % 10;
        result += pow(remainder, 3);
        originalNum /= 10;
    }
    if (result == num) {
        printf("%d is an Armstrong number.", num);
    } else {
        printf("%d is not an Armstrong number.", num);
    }
    return 0;
}

#include <stdio.h>
int main() {
    float basicSalary, homeAllowance, medicalAllowance, totalSalary;
    printf("Enter basic salary: ");
    scanf("%f", &basicSalary);
    if (basicSalary > 20000) {
        homeAllowance = 0.4 * basicSalary;
        medicalAllowance = 1500;
    } else {
        homeAllowance = 0.55 * basicSalary;
        medicalAllowance = 1000;
    }
    totalSalary = basicSalary + homeAllowance + medicalAllowance;
    printf("Total salary: %.2f\n", totalSalary);
    return 0;
}

int main() {
    char ch;
    printf("Enter a character: ");
    scanf(" %c", &ch);
    switch (ch) {
        case 'a':
        case 'e':
        case 'i':
        case 'o':
        case 'u':
        case 'A':
        case 'E':
        case 'I':
        case 'O':
        case 'U':
            printf("%c is a vowel.\n", ch);
            break;
        default:
            printf("%c is a consonant.\n", ch);
    }
    return 0;
}

#include <stdio.h>
int main() {
    int num, sum = 0, count = 0;
    printf("Enter a number (enter a negative number to terminate): ");
    scanf("%d", &num);
    while (num >= 0) {
        sum += num;
        count++;
        printf("Enter a number (enter a negative number to terminate): ");
        scanf("%d", &num);
    }
    float average = (float)sum / count;
    printf("The sum of the numbers is %d\n", sum);
    printf("The average of the numbers is %.2f\n", average);
    return 0;
}

int main() {
    int n, num, sum = 0;
    float average;
    printf("Enter the number of numbers: ");
    scanf("%d", &n);
    do {
        printf("Enter a number: ");
        scanf("%d", &num);
        sum += num;
    } while (--n > 0);
    average = (float)sum / n;
    printf("The average of the numbers is %.2f\n", average);
    return 0;
}
